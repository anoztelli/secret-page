                              def is_face_within_range(face_location, frame_shape):
                                  frame_center_x, frame_center_y = frame_shape[1] // 2, frame_shape[0] // 2
                                  face_center_x, face_center_y = face_location[0] + face_location[2] // 2, face_location[1] + face_location[3] // 2
                                  distance = ((frame_center_x - face_center_x) ** 2 + (frame_center_y - face_center_y) ** 2) ** 0.5
                                  return distance < FACE_RANGE_THRESHOLD

                              def lock_screen():

                              while True:
                                  ret, frame = cap.read()
                              
                                  if not ret:
                                      break
                              
                                  # Convert the frame to grayscale for face detection
                                  gray_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY)
                              
                                  # Detect faces in the grayscale frame
                                  faces = detector(gray_frame)
                              
                                  if len(faces) > 0:
                                      face = faces[0]  # Assuming only one face is present in the frame
                                      x, y, w, h = face.left(), face.top(), face.width(), face.height()
                              
                                      # Draw a rectangle around the detected face
                                      cv2.rectangle(frame, (x, y), (x + w, y + h), (0, 255, 0), 2)
                              
                                      # Check if the face is within range
                                      if not is_face_within_range((x, y, w, h), frame.shape):

                                  if cv2.waitKey(1) & 0xFF == ord('q'):
